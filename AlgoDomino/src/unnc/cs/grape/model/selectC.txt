void SelectSort(int list[],int n)
{
    int i,j,k,tmp;
    for(i=1;i<=n;i++)
    {
        k=i;
        for(j=i+1;j<=n;j++)
        {
            if(list[j]<list[k])
            {
                k=j;
            }
        }
        tmp=list[i];
        list[i]=list[k];
        list[k]=tmp;
    }
}
