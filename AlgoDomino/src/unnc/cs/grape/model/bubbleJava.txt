public void bubbleSort(int[] list) {
    for (int i = 0; i < list.length - 1; i++)
    {
        for (int j = list.length - 1; j > i; j--)
        {
            if (list[j - 1] > list[j])
            {
                int temp = list[j - 1];                
                list[j - 1] = list[j];               
                list[j] = temp;
            }
        }    
    }
}
